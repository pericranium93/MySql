  FROM users

-- Агрегация данных Задание 1
select 
	avg(floor((to_days(now()) - to_days(birthday_at)) / 365.25))
from users;

-- Задание 2
select
  date_format(date(concat_ws('-', year(now()), month(birthday_at), day(birthday_at))), '%W') as day_of_week,
  count(*) as total
from
  users
group by
  day_of_week
order by
  total desc;
